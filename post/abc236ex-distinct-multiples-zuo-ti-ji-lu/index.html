那么直接 dp 即可，时间复杂度 O(3n)O(3^n)O(3n)，代码如下：

#include <iostream>
#include <cstdio>

using namespace std;

typedef long long ll;

const int S=20,BS=1<<16,p=998244353;

int n;
ll m,a[S];
ll lcm[BS];
int f[S],dp[BS];

#define popc __builtin_popcount

inline ll gcd(ll x,ll y)
{
	if(x==0||y==0) return x+y;
	ll t=x%y;
	while(t!=0) x=y,y=t,t=x%y;
	return y;
}

inline ll getlcm(int st)
{
	ll res=1;
	for(int i=1;i<=n;i++)
	{
		if(st>>i-1&1)
		{
			ll g=gcd(res,a[i]);
			res/=g;
			if(res>m/a[i]) res=m+1;
			else res*=a[i];
		}
	}
	return res;
}

int main()
{
	scanf("%d%lld",&n,&m);
	for(int i=1;i<=n;i++) scanf("%lld",&a[i]);
	for(int i=0;i<(1<<n);i++) lcm[i]=getlcm(i);
	f[1]=1;
	for(int i=2;i<=n;i++) f[i]=p-1ll*(i-1)*f[i-1]%p;
	dp[0]=1;
	for(int i=1;i<(1<<n);i++)
	{
		for(int j=i;j>0;j=(j-1)&i)
		{
			if((i&-i)!=(j&-j)) continue;
			dp[i]=(dp[i]+1ll*f[popc(j)]*((m/lcm[j])%p)%p*dp[i^j]%p)%p;
		}
	}
	printf("%d\n",dp[(1<<n)-1]);
	return 0;
}
