// Problem: #187. count
// Contest: Hydro
// URL: http://oiclass.com/d/AKNOI/p/187
// Memory Limit: 512 MB
// Time Limit: 3000 ms
// 
// Powered by CP Editor (https://cpeditor.org)

#include <iostream>
#include <cstdio>

using namespace std;

const int MS=50005,S=500005;
const int p=998244353,ginv=332748118;

inline int qpow(int x,int y)
{
	int res=1;
	for(;y>0;y>>=1,x=1ll*x*x%p) res=y&1?1ll*res*x%p:res;
	return res;
}
inline int getlen(int n)
{
	int res=1;
	while(res<n) res<<=1;
	return res;
}
int lst_n,lst_rev[S];
inline void NTT(int n,int a[],int tpe)
{
	if(lst_n!=n)
	{
		lst_n=n;
		for(int i=0;i<n;i++) lst_rev[i]=(lst_rev[i>>1]>>1)|(i&1)*(n>>1);
	}
	for(int i=0;i<n;i++) if(lst_rev[i]<i) swap(a[i],a[lst_rev[i]]);
	int g=tpe==1?3:ginv;
	for(int mid=1;mid<n;mid<<=1)
	{
		int len=mid<<1,Wn=qpow(g,(p-1)/len);
		for(int l=0;l<n-len+1;l+=len)
		{
			for(int k=0,Wk=1;k<mid;k++,Wk=1ll*Wk*Wn%p)
			{
				int x=a[l+k],y=1ll*a[l+mid+k]*Wk%p;
				a[l+k]=(x+y)%p,a[l+mid+k]=(x-y+p)%p;
			}
		}
	}
}
inline void DFT(int n,int a[]){NTT(n,a,1);}
inline void IDFT(int n,int a[])
{
	NTT(n,a,-1);
	int inv=qpow(n,p-2);
	for(int i=0;i<n;i++) a[i]=1ll*a[i]*inv%p;
}

int f[S];
int h[S],g[S];
int q,n,a[MS];

void calc(int l,int r)
{
	if(r<=3||l>r) return;
	if(l==r) return f[l]=(f[l]+1ll*(l-2)*f[l-1]%p)%p,void();
	int mid=l+r>>1;
	calc(l,mid);
	
	// printf("calc([%d %d] -> %d):\n",l,r,mid);
	for(int i=l;i<=mid;i++) h[i-l]=i<3?0:f[i];
	for(int i=0;i<=r-l-1;i++) g[i]=1ll*i*f[i+2]%p;
	int lh=mid-l,lg=r-l-1;
	// for(int i=0;i<=lh;i++) printf("%d ",h[i]);
	// printf("\n");
	// for(int i=0;i<=lg;i++) printf("%d ",g[i]);
	// printf("\n");
	int rl=getlen(lh+lg+1);
	for(int i=lh+1;i<rl;i++) h[i]=0;
	for(int i=lg+1;i<rl;i++) g[i]=0;
	DFT(rl,h),DFT(rl,g);
	for(int i=0;i<rl;i++) g[i]=1ll*h[i]*g[i]%p;
	IDFT(rl,g);
	for(int i=mid+1;i<=r;i++) f[i]=(f[i]+g[i-l-1])%p;
	// printf("res: ");
	// for(int i=mid+1;i<=r;i++) printf("%d:%d ",i-l-1,g[i-l-1]);
	// printf("\n\n");
	
	for(int i=l-2;i<=mid-2;i++) h[i-l+2]=1ll*i*f[i+2]%p;
	for(int i=0;i<=r-l+1;i++) g[i]=i<3||i>=l?0:f[i];
	lh=mid-l,lg=r-l+1;
	// for(int i=0;i<=lh;i++) printf("%d ",h[i]);
	// printf("\n");
	// for(int i=0;i<=lg;i++) printf("%d ",g[i]);
	// printf("\n");
	rl=getlen(lh+lg+1);
	for(int i=lh+1;i<rl;i++) h[i]=0;
	for(int i=lg+1;i<rl;i++) g[i]=0;
	DFT(rl,h),DFT(rl,g);
	for(int i=0;i<rl;i++) g[i]=1ll*h[i]*g[i]%p;
	IDFT(rl,g);
	for(int i=mid+1;i<=r;i++) f[i]=(f[i]+g[i-l+1])%p;
	// printf("res: ");
	// for(int i=mid+1;i<=r;i++) printf("%d ",g[i-l+1]);
	// printf("\n***************\n");
	
	calc(mid+1,r);
}

int dfs(int r)
{
	int l=r-a[r]+1,res=1;
	int cnt=0;
	for(int i=r-1;i>=l;i--)
	{
		if(i-a[i]+1<l) return 0;
		res=1ll*res*dfs(i)%p;
		cnt++;
		i=i-a[i]+1;
	}
	res=1ll*res*f[cnt+1]%p;
	return res;
}

int main()
{
	freopen("count.in","r",stdin);
	freopen("count.out","w",stdout);
	f[1]=1,f[2]=2,f[3]=2;
	// for(int i=4;i<=100;i++)
	// {
		// f[i]=1ll*(i-2)*f[i-1]%p;

		// {
			// f[i]=(f[i]+1ll*h[j]*g[i-1-j]%p)%p;
		// }
	// }
	calc(1,MS-3);
	// for(int i=1;i<=20;i++) printf("%d ",f[i]);
	// printf("\n");
	scanf("%d%d",&q,&n);
	while(q--)
	{
		for(int i=1;i<=n;i++) scanf("%d",&a[i]);
		if(a[n]!=n) puts("0");
		else printf("%d\n",dfs(n));
	}
	return 0;
}
